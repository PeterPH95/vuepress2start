  // å°†é¡µé¢é‡Œçš„å­èŠ‚ç‚¹å¾ªç¯æ”¾å…¥æ–‡æ¡£ç¢ç‰‡
  while ((child = vm.$el.firstChild)) {
    fragment.appendChild(child);
  }
  fragment_compile(fragment);
  // æ›¿æ¢fragmenté‡Œæ–‡æœ¬èŠ‚ç‚¹çš„å†…å®¹
  function fragment_compile(node) {
    // ä½¿ç”¨æ­£åˆ™è¡¨è¾¾å¼å»åŒ¹é…å¹¶æ›¿æ¢èŠ‚ç‚¹é‡Œçš„{{}}
    const pattern = /\{\{\s*(\S+)\s*\}\}/;
    // 1. åŒ¹é…æ–‡æœ¬èŠ‚ç‚¹
    if (node.nodeType === 3) {
      // æå‰ä¿å­˜æ–‡æœ¬å†…å®¹ å¦åˆ™æ–‡æœ¬åœ¨è¢«æ›¿æ¢ä¸€æ¬¡å åç»­çš„æ“ä½œéƒ½ä¼šä¸ç”Ÿæ•ˆ
      // æ‰“å·¥äºº: {{name}}  => æ‰“å·¥äººï¼šè¥¿ç»´ å¦‚æœä¸ä¿å­˜åç»­ä¿®æ”¹nameä¼šåŒ¹é…ä¸åˆ°{{name}} å› ä¸ºå·²ç»è¢«æ›¿æ¢
      const texts = node.nodeValue;
      // è·å–æ­£åˆ™è¡¨è¾¾å¼åŒ¹é…æ–‡æœ¬å­—ç¬¦ä¸²è·å¾—çš„æ‰€æœ‰ç»“æœ
      const result_regex = pattern.exec(node.nodeValue);
      if (result_regex) {
        const arr = result_regex[1].split("."); // more.salary => ['more', 'salary']
        // ä½¿ç”¨reduceå½’å¹¶è·å–å±æ€§å¯¹åº”çš„å€¼ = vm.$data['more'] => vm.$data['more']['salary']
        const value = arr.reduce((total, current) => total[current], vm.$data);
        // ä¸‹é¢çš„è¿™ä¸€æ­¥å·²ç»æ›¿æ¢äº†å†…å®¹
        node.nodeValue = texts.replace(pattern, value);


        // åœ¨èŠ‚ç‚¹å€¼æ›¿æ¢å†…å®¹æ—¶ å³æ¨¡æ¿è§£æçš„æ—¶å€™ æ·»åŠ è®¢é˜…è€…
        // é€šè¿‡é—­åŒ…çš„æ–¹å¼è·å– node.nodeValue ä»è€Œåœ¨å°†æ¥ä½¿ç”¨ setter ä¿®æ”¹å±æ€§å€¼æ—¶è§¦å‘è§†å›¾æ›´æ–°
        new Watcher(vm, result_regex[1], (newVal) => {
          node.nodeValue = texts.replace(pattern, newVal);
        });
      }
    }
    // 2. åŒ¹é… input èŠ‚ç‚¹
    // æ›¿æ¢ç»‘å®šäº†v-modelå±æ€§çš„inputèŠ‚ç‚¹çš„å†…å®¹
    if (node.nodeType === 1 && node.nodeName === "INPUT") {
      const attr = Array.from(node.attributes);
      attr.forEach((item) => {
        if (item.nodeName === "v-model") {
          const value = item.nodeValue
            .split(".")
            .reduce((total, current) => total[current], vm.$data);
          node.value = value;

          // ä¿è¯å°†æ¥ä½¿ç”¨ setter ä¿®æ”¹å±æ€§å€¼æ—¶è§¦å‘è§†å›¾æ›´æ–°
          new Watcher(vm, item.nodeValue, (newVal) => {
            node.value = newVal;
          });

          // é€šè¿‡ç»‘å®šäº‹ä»¶å®ç° View -> VM çš„æ›´æ–°
          node.addEventListener("input", (e) => {
            // ['more', 'salary']
            const arr1 = item.nodeValue.split(".");
            // ['more']
            const arr2 = arr1.slice(0, arr1.length - 1);
            // vm.$data.more
            const final = arr2.reduce(
              (total, current) => total[current],
              vm.$data
            );
            // vm.$data.more['salary'] = e.target.value
            final[arr1[arr1.length - 1]] = e.target.value;
          });
        }
      });
    }
    // å¯¹å­èŠ‚ç‚¹çš„æ‰€æœ‰å­èŠ‚ç‚¹ä¹Ÿè¿›è¡Œæ›¿æ¢å†…å®¹æ“ä½œ
    node.childNodes.forEach((child) => fragment_compile(child));
  }
  // æ“ä½œå®Œæˆåå°†æ–‡æ¡£ç¢ç‰‡æ·»åŠ åˆ°é¡µé¢
  // æ­¤æ—¶å·²ç»èƒ½å°†vmçš„æ•°æ®æ¸²æŸ“åˆ°é¡µé¢ä¸Š ä½†è¿˜æœªå®ç°æ•°æ®å˜åŠ¨çš„åŠæ—¶æ›´æ–°
  vm.$el.appendChild(fragment);
}

  constructor() {

  }

  // éœ€è¦vueå®ä¾‹ä¸Šçš„å±æ€§ ä»¥è·å–æ›´æ–°ä»€ä¹ˆæ•°æ®
  constructor(vm, key, callback) {

    this.callback = callback;// è®°å½•å¦‚ä½•æ›´æ–°æ–‡æœ¬å†…å®¹
    //1. ä¸´æ—¶å±æ€§ ä¿å­˜ä¸€ä¸ª Watcher å®ä¾‹
    Dependency.temp = this;

    key.split(".").reduce((total, current) => total[current], vm.$data);// è¯»å–å±æ€§è§¦å‘getter --> vm.$data['more']['salary']
    //3. ç½®ç©ºï¼Œé˜²æ­¢è®¢é˜…è€…å¤šæ¬¡åŠ å…¥åˆ°ä¾èµ–å®ä¾‹æ•°ç»„é‡Œ
    Dependency.temp = null;

  update() {
    const value = this.key

    this.callback(value);
  }
